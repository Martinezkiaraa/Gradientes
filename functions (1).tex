\begin{document}
function:
\[
  f = ((\tanh(w^\top \cdot x+b)+1)/2-d)^{2}
\]

gradient:
\[
  \frac{\partial f}{\partial b} = (1-\tanh(b+w^\top \cdot x)^{2})\cdot ((1+\tanh(b+w^\top \cdot x))/2-d)
\]
where
\begin{itemize}
  \item $b$ is a scalar
  \item $d$ is a scalar
  \item $w$ is a vector
  \item $x$ is a vector
\end{itemize}
